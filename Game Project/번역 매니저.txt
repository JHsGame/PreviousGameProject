    public void OnCheckImage(int _value)
    {

            SoundManager_sfx.instance.PlaySE("UI_Click1", false);

        i_LanguageValue = _value;
        g_BeforeDropDown.GetComponent<Dropdown>().value = i_LanguageValue;
        g_AfterDropDown.GetComponent<Dropdown>().value = i_LanguageValue;

        if(i_LanguageValue == 2 || i_LanguageValue == 3)
        {
            ChangeFont(g_BeforeDropDown.GetComponent<Dropdown>().captionText);
            ChangeFont(g_AfterDropDown.GetComponent<Dropdown>().captionText);
